#include <tunables/global>

profile tsun-proxy-dev flags=(attach_disconnected,mediate_deleted,audit) {

  # S6-Overlay benötigt diese Capabilities, um Dienste zu starten
  capability chown,
  capability setuid,
  capability setgid,
  capability dac_override,
  capability dac_read_search,
  capability fowner,
  capability fsetid,
  capability kill,
  # Capabilities fuer Signale
  signal (send,receive) set=(kill,term,int,hup,cont),

  # Der Übergang (ohne Includes im Hauptteil)
  /usr/bin/python3* r,
  /usr/bin/python3* cx -> myprogram,

  # S6-Overlay Kernel-Rights
  /init r,
  /init ix,
  /dev/null rw,
  /dev/urandom r,
  /dev/tty rw,
  /dev/console rw,
  /dev/pts/* rw,
  /dev/pts/[0-9]* rw,
  /dev/fd/ r,
  /dev/fd/** rw,
  /proc/*/fd/ r,
  /proc/*/fd/** rw,
  /proc/self/fd/ r,
  /proc/self/fd/** rw,

  /bin/** ix,
  /command/** r,
  /command/** ix,
  /package/** r,
  /package/** ix,
  
  # Netzwerk-Zugriff für Bashio/Curl (Supervisor API Check)
  network inet stream,
  network inet6 stream,
  network inet dgram,
  network inet6 dgram,

  # Wichtige Dateien für die Namensauflösung (DNS)
  /etc/resolv.conf r,
  /etc/hosts r,
  /etc/nsswitch.conf r,
  /etc/ssl/certs/ca-certificates.crt r, # Falls HTTPS genutzt wird
  /etc/ssl/certs/ r,

  # /etc Zugriff (S6-Konfigurationen)
  /etc/ r,
  /etc/s6-overlay/ r,
  /etc/s6-overlay/** r,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  

  # Runtime & Verzeichnisse (l für Verknüpfungen/Erstellung)
  /run/ w,
  /run/** rwk,
  /run/{s6,s6-rc*,service}/ rw,
  /run/{s6,s6-rc*,service}/** rwk,
  /run/{s6,s6-rc*,service}/** ix,
  /run/s6-rc/servicedirs/ w,
  /run/s6-rc/servicedirs/** rwk,
  /run/s6-rc/servicedirs/** l,
  /run/uncaught-logs/ w,
  /run/uncaught-logs/** rwk,

  # S6-Kommunikation (FIFOs und Sockets)
  /run/s6-rc:*/ r,
  /run/s6-rc:**/ rwk,


  # S6-Overlay Binaries (often Symlinks)
  /usr/bin/bash ix,
  /usr/bin/s6-* ix,
  /usr/bin/env ix,
  /usr/bin/[^p]* ix,   # Erlaubt fast alles, was nicht mit p anfängt (z.B. sed, grep)
  /usr/bin/p[^y]* ix,  # Erlaubt Programme mit p, aber nicht py...
  /usr/lib/bashio/** ix,
  /data/** rw,
  /tmp/** rwk,
  
  # Bibliotheken und der Linker (Essenziell für Bash und S6)
  /lib/** r,
  /lib/** mr,
  /usr/lib/** r,
  /usr/lib/** mr,
  
  # Spezifisch für den dynamischen Linker
  /lib/ld-*.so* mr,
  /lib/*/ld-*.so* mr,
  /usr/lib/ld-*.so* mr,


  /run.sh r,
  /run.sh ix,
  /proxy-version.txt r,
  /home/proxy/server.py r,

  # Erlaubt das Auflisten von Verzeichnissen
  / r,
  /**/ r,  # Erlaubt das Lesen (Listing) aller Verzeichnisse

  profile myprogram flags=(attach_disconnected,mediate_deleted,audit) {

    # 1. System & Libs (Der Segfault-Schutz)
    /usr/lib/** mr,
    /lib/** mr,
    /usr/local/lib/** mr,
    /usr/bin/python3* ix,
    /etc/ld.so.cache r,

    # 2. App & HA Schnittstellen
    /home/proxy/ r,
    /home/proxy/** r,
    /home/translations/** r,
    /homeassistant/tsun-proxy/** rw,
    /data/** rw,             # Access to options.json and other files within your addon

    # 3. System-Info & Kernel
    /proc/self/** r,
    /proc/cpuinfo r,
    /sys/** r,               # Network-Test: Needed for Hw and platform detection
    
    # 4. Standard Hardware / Logs
    /dev/urandom r,
    /dev/tty rw,

    # 5. Netzwerk (aus der nameservice Abstraction unterstützt)
    network inet stream,
    network inet dgram,

    # 6. Sonstiges (Wurzelverzeichnis-Metadaten)
    /usr/** r,
    /etc/** r,
    
    # Signale vom Hauptprofil empfangen
    signal (receive) peer=*tsun-proxy-dev,
  }
}
