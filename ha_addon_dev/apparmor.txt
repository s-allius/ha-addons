#include <tunables/global>

profile tsun-proxy-dev flags=(attach_disconnected,mediate_deleted,audit) {

  # S6-Overlay benötigt diese Capabilities, um Dienste zu starten
  capability chown,
  capability setuid,
  capability setgid,
  capability dac_override,
  capability dac_read_search,
  capability fowner,
  capability fsetid,
  capability kill,

  # Der Übergang (ohne Includes im Hauptteil)
  /usr/bin/python3* r,
  /usr/bin/python3* cx -> myprogram,

  # S6-Overlay Kernel-Rights
  /init r,
  /init ix,
  /dev/null rw,
  /dev/urandom r,
  /dev/tty rw,
  /dev/console rw,
  
  /bin/** ix,
  /command/** r,
  /command/** ix,
  /package/** r,
  /package/** ix,
  
  # Netzwerk-Zugriff für Bashio/Curl (Supervisor API Check)
  network inet stream,
  network inet6 stream,
  network inet dgram,
  network inet6 dgram,

  # Wichtige Dateien für die Namensauflösung (DNS)
  /etc/resolv.conf r,
  /etc/hosts r,
  /etc/nsswitch.conf r,
  /etc/ssl/certs/ca-certificates.crt r, # Falls HTTPS genutzt wird
  /etc/ssl/certs/ r,

  # /etc Zugriff (S6-Konfigurationen)
  /etc/ r,
  /etc/s6-overlay/ r,
  /etc/s6-overlay/** r,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,

  # Runtime & Verzeichnisse (l für Verknüpfungen/Erstellung)
  /run/ w,
  /run/** rwk,
  /run/{s6,s6-rc*,service}/ rw,
  /run/{s6,s6-rc*,service}/** rwk,
  /run/{s6,s6-rc*,service}/** ix,
  /run/s6-rc/servicedirs/ w,
  /run/s6-rc/servicedirs/** rwk,
  /run/s6-rc/servicedirs/** l,

  # S6-Kommunikation (FIFOs und Sockets)
  /run/s6-rc:*/ r,
  /run/s6-rc:**/ rwk,


  # S6-Overlay Binaries (often Symlinks)
  /usr/bin/bash ix,
  /usr/bin/s6-* ix,
  /usr/bin/env ix,
  /usr/bin/[^p]* ix,   # Erlaubt fast alles, was nicht mit p anfängt (z.B. sed, grep)
  /usr/bin/p[^y]* ix,  # Erlaubt Programme mit p, aber nicht py...
  /usr/lib/bashio/** ix,
  /data/** rw,
  /tmp/** rwk,
  
  # Bibliotheken und der Linker (Essenziell für Bash und S6)
  /lib/** r,
  /lib/** mr,
  /usr/lib/** r,
  /usr/lib/** mr,
  
  # Spezifisch für den dynamischen Linker
  /lib/ld-*.so* mr,
  /lib/*/ld-*.so* mr,
  /usr/lib/ld-*.so* mr,

  # Capabilities fuer Signale
  signal (send) set=(kill,term,int,hup,cont),

  /run.sh r,
  /run.sh ix,
  /proxy-version.txt r,
  /home/proxy/server.py r,

  profile myprogram flags=(attach_disconnected,mediate_deleted) {
    # --- Das Minimum fuer Python ---
    
    # Der Interpreter selbst
    /usr/bin/python3* ix,
    
    # Libraries (Shared Objects & Python Modules)
    /lib/** r,
    /usr/lib/** r,
    /usr/local/lib/** r,
    
    # Dynamischer Linker (Essenziell, damit Python startet)
    /lib/ld-*.so* mr,
    /lib/*/ld-*.so* mr,

    # Deine App & Daten
    /home/proxy/ r,
    /home/proxy/** r,
    /data/** rw,
    /share/** rw,
    /dev/tty rw,

    # Netzwerk (Raw, statt Abstraction)
    network inet stream,
    network inet6 stream,
    network inet dgram,

    # System-Infos (DNS, Zeit, User)
    /etc/hosts r,
    /etc/resolv.conf r,
    /etc/nsswitch.conf r,
    /etc/passwd r,
    /etc/group r,
    /etc/ld.so.cache r,

    # Signale von S6 empfangen
    signal (receive) peer=ADDON_SLUG,
    
    # Randomness fuer Python (Dictionaries/Security)
    /dev/urandom r,
  }
}
