#include <tunables/global>

profile tsun-proxy-dev flags=(attach_disconnected,mediate_deleted) {

  # S6-Overlay benötigt diese Capabilities, um Dienste zu starten
  capability chown,
  capability setuid,
  capability setgid,
  capability dac_override,
  capability dac_read_search,
  capability fowner,
  capability fsetid,
  capability kill,
  # Capabilities fuer Signale
  signal (send,receive) set=(kill,term,int,hup,cont),

  # Der Übergang (ohne Includes im Hauptteil)
  /usr/bin/python3* r,
  /usr/bin/python3* cx -> myprogram,

  # S6-Overlay Kernel-Rights
  /init rix,
  /dev/null rw,
  /dev/urandom r,
  /dev/tty rw,

  /bin/** ix,
  /command/** ix,
  /package/** rix,
  
  # Netzwerk-Zugriff für Bashio/Curl (Supervisor API Check)
  network inet stream,
  network inet6 stream,
  network inet dgram,
  network inet6 dgram,

  /etc/** r,
  # /etc/services.d/** rwix,
  # /etc/cont-init.d/** rwix,
  # /etc/cont-finish.d/** rwix,
  /etc/services.d/** ix,
  /etc/cont-init.d/** ix,
  /etc/cont-finish.d/** ix,
  

  # Runtime & Verzeichnisse (l für Verknüpfungen/Erstellung)
  /run/ w,
  /run/** rwk,
  /run/{s6,s6-rc*,service}/** rwk,
  /run/{s6,s6-rc*,service}/** ix,


  # S6-Overlay Binaries (often Symlinks)
  /usr/bin/[^p]* ix,   # Erlaubt fast alles, was nicht mit p anfängt (z.B. sed, grep)
  /usr/bin/p[^y]* ix,  # Erlaubt Programme mit p, aber nicht py...
  /usr/lib/bashio/** ix,
  /data/** rw,
  /tmp/** rwk,
  
  # Bibliotheken und der Linker (Essenziell für Bash und S6)
  /lib/** mr,
  /usr/lib/** mr,
  
  # Spezifisch für den dynamischen Linker
  /lib/ld-*.so* mr,
  /lib/*/ld-*.so* mr,
  /usr/lib/ld-*.so* mr,


  /run.sh rix,
  /proxy-version.txt r,
  /home/proxy/server.py r,

  profile myprogram flags=(attach_disconnected,mediate_deleted) {

    # 1. System & Libs
    /usr/** r,
    /etc/** r,

    /usr/lib/** mr,
    /lib/** mr,
    /usr/local/lib/** mr,

    # 2. App & HA Schnittstellen
    /home/proxy/ r,
    /home/proxy/** r,
    /home/translations/** r,
    /homeassistant/tsun-proxy/** rw,
    /data/** rw,             # Access to options.json and other files within your addon

    # 3. System-Info & Kernel
    /proc/self/** r,
    /proc/cpuinfo r,
    /sys/** r,               # Network-Test: Needed for Hw and platform detection
    
    # 4. Standard Hardware / Logs
    /dev/urandom r,
    /dev/tty rw,

    # 5. Netzwerk (aus der nameservice Abstraction unterstützt)
    network inet stream,
    network inet dgram,

    # Signale vom Hauptprofil empfangen
    signal (receive) peer=*tsun-proxy-dev,
  }
}
