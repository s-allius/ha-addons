#include <tunables/global>

profile tsun-proxy-dev flags=(attach_disconnected,mediate_deleted) {
  # Der Übergang (ohne Includes im Hauptteil)
  # deny /usr/bin/python3* x,
  /usr/bin/python3* px -> myprogram,

  # --- Hauptprofil (S6/Bash) ---
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /usr/lib/bashio/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /etc/services.d/** rwix,
  /data/** rw,
  /tmp/** rwk,
  /dev/tty rw,
  
  # Capabilities fuer Signale
  capability kill,
  signal (send) set=(kill,term,int,hup,cont),


  profile myprogram flags=(attach_disconnected,mediate_deleted) {
    # --- Das Minimum fuer Python ---
    
    # Der Interpreter selbst
    /usr/bin/python3* ix,
    
    # Libraries (Shared Objects & Python Modules)
    /lib/** r,
    /usr/lib/** r,
    /usr/local/lib/** r,
    
    # Dynamischer Linker (Essenziell, damit Python startet)
    /lib/ld-*.so* mr,
    /lib/*/ld-*.so* mr,

    # Deine App & Daten
    /home/proxy/ r,
    /home/proxy/** r,
    /data/** rw,
    /share/** rw,
    /dev/tty rw,

    # Netzwerk (Raw, statt Abstraction)
    network inet stream,
    network inet6 stream,
    network inet dgram,

    # System-Infos (DNS, Zeit, User)
    /etc/hosts r,
    /etc/resolv.conf r,
    /etc/nsswitch.conf r,
    /etc/passwd r,
    /etc/group r,
    /etc/ld.so.cache r,

    # Signale von S6 empfangen
    signal (receive) peer=ADDON_SLUG,
    
    # Randomness fuer Python (Dictionaries/Security)
    /dev/urandom r,
  }
}
